def find_via_shortname!(url)
  if url.to_i.to_s == url.to_s
    self.find url.to_i
  else
    find_via_shortname_nil(url) or raise(ActiveRecord::RecordNotFound)
  end
end

def find_via_shortname_nil(url)

end

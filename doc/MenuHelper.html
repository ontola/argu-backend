def crud_menu_item(resource, additional_items = [])
  if current_user.present? && policy(resource).update?
    content_tag :li do
      content_tag :ul do
        react_component 'HyperDropdown',
                        crud_menu_options(resource, additional_items),
                        {prerender: true}
      end
    end
  end
end

def flow
  activities = Activity.arel_table
  Activity.where(
    activities[:trackable_id].eq(id).and(
      activities[:trackable_type].eq(model_name.to_s))
      .or(activities[:recipient_id].eq(self).and(
        activities[:recipient_type].eq(model_name.to_s))))
end

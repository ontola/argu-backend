def moderator=(value)
  if value.is_a?(String)
    value.is_a?(String)
    entity = User.find_via_shortname_nil(value)
    entity ||= Group.find_by(id: value)
    entity ||= Group.find_by(name: value)
    value = entity
  end
  if value.is_a?(Group)
    self.group = value
    self.user = nil
  elsif value.is_a?(User)
    self.user = value
    self.group = nil
  elsif value == nil
    nil
  else
    raise 'Unthought clause happened'
  end
end

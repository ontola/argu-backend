def trash
  self.class.transaction do
    if self.respond_to?(:trashed_at)
      update_column :trashed_at, DateTime.current
    else
      update_column :is_trashed, true
    end
    refresh_counter_cache if self.respond_to? :refresh_counter_cache
  end
end

def self.get(key, redis = self.redis_instance)
  begin
    redis.get(key)
  rescue ::Redis::CannotConnectError => e
    self.rescue_redis_connection_error(e)
  end
end

def self.rescue_redis_connection_error(e)
  Rails.logger.error 'Redis not available'
  ::Bugsnag.notify(e, {
                        :severity => 'error',
                    })
end

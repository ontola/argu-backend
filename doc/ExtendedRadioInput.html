def to_html
  input_wrapping do
    choices_wrapping do
      legend_html <<
          choices_group_wrapping do
            collection.map { |choice|
              choice_wrapping(choice_wrapping_html_options(choice)) do
                choice_html(choice) <<
                extended_hint_html(choice)
              end
            }.join("\n").html_safe
          end
    end
  end
end

def get_current_actor
  @_current_actor ||= if cookies[:a_a]
    p = Profile.find(cookies[:a_a])
    raise 'not authorized' unless ActorPolicy.new(UserContext.new(current_user, nil, session), p).show?
    p
  else
    current_user.profile
  end
end
